const compose = (f, g) => (x) => f(g(x))

const head = (x) => x[0]

const reverse = xs => xs.reverse()

const last = compose(head, reverse)

const list = [
  'Meetup', 
  'JavaScript', 
  'Litoral',
]
 
last(list) //=> 'Litoral'

head(reverse(list)) //=> 'Litoral'

reverse(head(list)) //=> throws

const pipe = (f, g) => (x) => g(f(x))

const last = pipe(reverse, head)

last(list) //=> 'Litoral
